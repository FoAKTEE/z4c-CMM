% Default figure macro. The required default width is 0.8\columnwidth.
\newcommand{\paperfig}[4][0.8\columnwidth]{%
\begin{figure}[htbp]
    \centering
    \includegraphics[width=#1]{#2}
    \caption{#3}
    \label{#4}
\end{figure}%
}

\newcommand{\papertable}[3]{%

\end{table}%
}
